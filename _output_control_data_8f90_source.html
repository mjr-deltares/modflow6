132  iprint = 1

134  endif
135  endif
136  !

138  if (present(isav_opt)) then
139  if (isav_opt /= 0) then
140  idataun = this%idataun

144  endif
145  !
146  ! -- Record double precision array

148  call this%dis%record_array(this%dblvec, iout, iprint, idataun, &
149  this%cname, this%cdatafmp, this%nvaluesp, &

313  character(len=*),parameter :: fmtocsave = &

316 ! ------------------------------------------------------------------------------
317  !
318  line(:) = linein(:)

320  call urword(line, lloc, istart, istop, 1, ival, rval, 0, 0)

324  this%idataun = getunit()
325  write(iout, fmtocsave) trim(adjustl(this%cname)), this%idataun, &
326  line(istart:istop)
327  call openfile(this%idataun, iout, line(istart:istop), 'DATA(BINARY)', &
328  form, access, 'REPLACE', mnormal)
329  case('PRINT_FORMAT')
330  call urword(line, lloc, istart, istop, 1, ival, rval, 0, 0)
331  call print_format(line(istart:), this%cdatafmp, this%editdesc, &
332  this%nvaluesp, this%nwidthp, inunit)
333  case default
334  call store_error('Looking for FILEOUT or PRINT_FORMAT. Found:')
335  call store_error(trim(adjustl(line)))
336  call store_error_unit(inunit)
