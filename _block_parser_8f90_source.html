173  subroutine getnextline(this, endOfBlock)
174 ! ******************************************************************************
175 ! GetNextLine -- read the next line
176 ! ******************************************************************************
177 !
178 ! SPECIFICATIONS:
179 ! ------------------------------------------------------------------------------
180  ! -- dummy
181  class(blockparsertype), intent(inout) :: this
182  logical, intent(out) :: endOfBlock
183  ! -- local
184  integer(I4B) :: ierr, ival
185  integer(I4B) :: istart, istop
186  real(DP) :: rval
187  character(len=10) :: key
188  logical :: lineread
189 ! ------------------------------------------------------------------------------
190  !
191  endofblock = .false.
192  ierr = 0
193  lineread = .false.
194  loop1: do
195  if (lineread) exit loop1
196  call u9rdcom(this%iuext, this%iout, this%line, ierr)
197  this%lloc = 1
198  call urword(this%line, this%lloc, istart, istop, 0, ival, rval, &
199  this%iout, this%iuext)
200  key = this%line(istart:istop)
201  call upcase(key)
202  if (key == 'END' .or. key == 'BEGIN') then
203  call uterminate_block(this%inunit, this%iout, key, &
204  this%blockNameFound, this%lloc, this%line, &
205  ierr, this%iuext)
206  this%iuactive = this%iuext
207  endofblock = .true.
208  lineread = .true.
209  elseif (key == '') then
210  ! End of file reached.
211  ! If this is an OPEN/CLOSE file, close the file and read the next
212  ! line from this%inunit.
213  if (this%iuext /= this%inunit) then
214  close(this%iuext)
215  this%iuext = this%inunit
216  this%iuactive = this%inunit
217  else
218  errmsg = 'Unexpected end of file reached.'
219  call store_error(errmsg)
220  call this%StoreErrorUnit()
221  call ustop()
222  endif
223  else
224  this%lloc = 1
225  this%linesRead = this%linesRead + 1
226  lineread = .true.
227  endif
228  enddo loop1
229  !
230  ! -- return
231  return
232  end subroutine getnextline
